1 Neopixel コード

下から順にランダムなところまで光って上から順に消えるコード
  1. #include <Adafruit_NeoPixel.h>
  2. #define PIN 9 // 信号用のピンを指定
  3. #define NUMPIXELS 5 // LEDの数を指定
  4. int brightness=50;//明るさ
  6. Adafruit_NeoPixel pixels(NUMPIXELS, PIN, NEO_GRB + NEO_KHZ800);
  8. void setup() {
  9.   pixels.begin(); // NeoPixel出力ピンの初期化
  10.   pixels.setBrightness(brightness);
  11. }
  13. void loop() {
  15.   for(int i=0; i<random(6); i++) {
  16.   int k=random(256);
  17.   int l=random(256);
  18.   int o=random(256);
  19.     pixels.setPixelColor(i, pixels.Color(k, 0, l));
  20.     pixels.show();
  21.     delay(100);
  22.     
  23.   }
  24.   for(int i=0; i<5; i++) {
  25.     pixels.setPixelColor(4-i, pixels.Color(0, 0, 0));
  26.     pixels.show();
  27.     delay(100);
  28.   }
  29. }
