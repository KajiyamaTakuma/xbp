1 ステッピングモーター コード

光センサーで回転が変わるコード
  1. const int DIR = 8;
  2. const int STEP = 9;
  4. void setup() {
  5.   pinMode(DIR, OUTPUT);
  6.   pinMode(STEP, OUTPUT);
  7.   digitalWrite(DIR, LOW);
  8.   digitalWrite(STEP, LOW);
  9.   Serial.begin(9600);
  10. }
  12. void loop() {
  13.   int val=0;
  14.   val=analogRead(1);
  15.   Serial.println(val);
  16.  if(val<200){
  17.       clockwise(5000);
  18.     }else{
  19.       
  20.        counter_clockwise(5000);
  21.     }
  22.   
  23.   }
  25. void clockwise(int delaytime){
  26.   digitalWrite(DIR, HIGH);//HIGHは時計回り
  27.   digitalWrite(STEP, HIGH);
  28.   delayMicroseconds(delaytime);
  29.   digitalWrite(STEP, LOW);
  30.   delayMicroseconds(delaytime);
  32. }
  34. void counter_clockwise(int delaytime){
  36.   digitalWrite(DIR, LOW);//はLOWは反時計回り
  37.   digitalWrite(STEP, HIGH);
  38.   delayMicroseconds(delaytime);
  39.   digitalWrite(STEP, LOW);
  40.   delayMicroseconds(delaytime);
  41. }
