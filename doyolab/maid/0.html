gitpush.py

  1. #!/usr/bin/env python3
  2. # -*- coding: utf-8 -*-
  4. import os
  5. import subprocess
  6. from datetime import datetime
  7. import tkinter as tk
  8. from tkinter import filedialog, messagebox
  11. def execute_git_commands(repo_directory):
  12.     os.chdir(repo_directory)
  14.     subprocess.run(['git', 'status'])
  15.     subprocess.run(['git', 'add', '.'])
  17.     current_date = datetime.now().strftime("%Y-%m-%d")
  18.     commit_message = current_date
  20.     subprocess.run(['git', 'commit', '-m', commit_message])
  21.     subprocess.run(['git', 'push'])
  23.     messagebox.showinfo("完了", "Gitコマンドの実行が完了しました。")
  26. def browse_directory():
  27.     directory = filedialog.askdirectory()
  28.     repo_directory.set(directory)
  31. def run_script():
  32.     execute_git_commands(repo_directory.get())
  35. root = tk.Tk()
  36. root.title("Git Automation")
  38. repo_directory = tk.StringVar()
  40. frame = tk.Frame(root)
  41. frame.pack(padx=20, pady=20)
  43. label = tk.Label(frame, text="リポジトリのディレクトリ:")
  44. label.grid(row=0, column=0)
  46. entry = tk.Entry(frame, textvariable=repo_directory, width=40)
  47. entry.grid(row=0, column=1)
  49. browse_button = tk.Button(frame, text="参照", command=browse_directory)
  50. browse_button.grid(row=0, column=2)
  52. run_button = tk.Button(frame, text="実行", command=run_script)
  53. run_button.grid(row=1, column=1, pady=20)
  55. root.mainloop()
